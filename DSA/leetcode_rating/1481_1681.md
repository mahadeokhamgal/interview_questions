def minimumLines(self, stockPrices: List[List[int]]) -> int:
    stockPrices.sort()
    lines = 0
    prev_dx, prev_dy = None, None

    for i in range(1, len(stockPrices)):
        x2, y2 = stockPrices[i]
        x1, y1 = stockPrices[i-1]
        dx = x2-x1
        dy = y2-y1
        if i == 1 or dx*prev_dy != dy*prev_dx:
            lines +=1
            prev_dx, prev_dy = dx, dy

    return lines
```

# 1865. https://leetcode.com/problems/finding-pairs-with-a-certain-sum/ Rating - 1680

class FindSumPairs:

    def __init__(self, nums1: List[int], nums2: List[int]):
        self.counter1 = Counter(nums1)
        self.counter2 = Counter(nums2)
        self.nums2 = nums2

    def add(self, index: int, val: int) -> None:
        self.counter2[self.nums2[index]] -= 1
        self.nums2[index] += val
        self.counter2[self.nums2[index]] += 1

    def count(self, tot: int) -> int:
        ans = 0
        for key1 in self.counter1.keys():
            key2 = tot - key1
            if self.counter1[key1] > 0 and self.counter2[key2] > 0:
                ans += self.counter1[key1] * self.counter2[key2]

        return ans
```

def maxSumTwoNoOverlap(self, nums: List[int], firstLen: int, secondLen: int) -> int:
    """likely LTE"""
    N = len(nums)
    maxLen = max(firstLen, secondLen)
    minLen = min(firstLen, secondLen)
    maxPrefixSum = [-1] * N
    runningSum = sum(nums[:maxLen-1])
    
    for i in range(maxLen-1, N):
        runningSum += nums[i]
        if i >= maxLen:
            runningSum -= nums[i-maxLen]
        
        maxPrefixSum[i] = runningSum
    
    runningSum = sum(nums[:minLen-1])
    ans = 0

    for i in range(minLen-1, N):
        runningSum += nums[i]
        if i >= minLen:
            runningSum -= nums[i-minLen]
            
        maxOther = 0
        if i-minLen >= 0:
            maxOther = max(maxOther, max(maxPrefixSum[:i-minLen+1]))
        if i+maxLen < N:
            maxOther = max(maxOther, max(maxPrefixSum[i+maxLen:]))
        ans = max(ans, runningSum + maxOther)
        
    return ans
